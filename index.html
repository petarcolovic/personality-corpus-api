Personality Corpus ‚Äì Concept Viewer
Browse lemmas grouped by language for one kernel concept.
Kernels are loaded from /kernels. Languages are loaded from /languages. Data per kernel comes from /lemmas/by_kernel/<kernel_word>.
API docs: /docs
async function searchLemmas() { clearError(); const kernel = kernelSelect.value; const langFilter = languageSelect.value; if (!kernel) { showError("Please choose a kernel word first."); return; } try { const encodedKernel = encodeURIComponent(kernel); const url = `${API_BASE}/lemmas/by_kernel/${encodedKernel}?page=1&page_size=1000`; const data = await fetchJSON(url); const rows = Array.isArray(data) ? data : (data.results || []); if (!Array.isArray(rows) || rows.length === 0) { resultsEl.style.display = "block"; resultsMetaEl.innerHTML = "No lemmas found for this kernel."; resultsBodyEl.innerHTML = ""; return; } // Grupisanje po jeziku const byLang = {}; rows.forEach((item) => { const langObj = item.language || {}; const prefix = langObj.prefix || item.lang_prefix || ""; const name = langObj.name || ""; const langKey = prefix || name || "Unknown"; if (!byLang[langKey]) { byLang[langKey] = { language: name || langKey, language_prefix: prefix, lemmas: [], }; } const labelParts = []; if (item.word_original) { labelParts.push(item.word_original); } if (item.word_en) { labelParts.push("[" + item.word_en + "]"); } const label = labelParts.join(" "); byLang[langKey].lemmas.push(label || "(no label)"); }); const groups = Object.values(byLang); let totalLemmas = 0; groups.forEach((g) => { totalLemmas += g.lemmas.length; }); resultsEl.style.display = "block"; resultsMetaEl.innerHTML = ` Kernel: ${kernel} ${groups.length} language(s),  ${totalLemmas} lemma(s) `; resultsBodyEl.innerHTML = ""; groups.forEach((g) => { const div = document.createElement("div"); div.className = "lemma-group"; const lemmasText = g.lemmas; div.innerHTML = `
${g.language} ${ g.language_prefix ? ` (${g.language_prefix})` : "" }
${lemmasText.length} lemma(s)
${lemmasText .map((lemma) => `${lemma}`) .join("")}
`; resultsBodyEl.appendChild(div); }); } catch (err) { showError("Error loading lemmas. " + err.message); } } document.addEventListener("DOMContentLoaded", () => { loadLanguages(); loadKernels(); searchBtn.addEventListener("click", searchLemmas); });
